\subsubsection*{\texttt{Email}}
Semua peubah \texttt{Email} berjenis \texttt{string}.
Berikut ini gambaran deklarasi peubah-peubah kelasnya.
\begin{verbatim}
class Email
{
private:
  string senderAddr;
  string recipientAddr;
  string subject;
  string content;
};
\end{verbatim}
